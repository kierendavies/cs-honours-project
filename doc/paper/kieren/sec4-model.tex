\[
  \test_O(A) =
  \begin{cases}
    \mathrm{inconsistent} &
      \text{if } O \union A \sementails \bot \\
    \mathrm{incoherent} &
      \text{if } \exists C (O \sementails C \sqsubseteq \nothing) \\
    \mathrm{absent} &
      \text {if } O \not\sementails A \\
    \mathrm{entailed} &
      \text{if } O \sementails A \\
  \end{cases}
